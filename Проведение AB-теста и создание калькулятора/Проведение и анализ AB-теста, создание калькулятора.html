def test_calc (r1, r2, alpha = 0.05):
    s, p = ttest_ind(r1, r2)
    if p>alpha:
        print(f"Итоги теста:\nalpha = {alpha}\npval = {p}\nt-кретерий = {s}\
\nНет оснований для отклонения нулевой гипотезы. Нельзя сказать, что значения конверсий различно.\
\nСтатистически значимой разницы в конверсии не обнаружено, так как p-value выше выбранного порога ошибки первого рода.")
    else:
        print(f"Итоги теста:\nalpha = {alpha}\npval = {p}\nt-кретерий = {s}\
\nНеобходимо отвергнуть нулевую гепотизу.\
\nСтатистически значимая разница в конверсии обнаружена, так как p-value ниже выбранного порога ошибки первого рода.")
        
    if (s <scipy.stats.norm.ppf(1-alpha/2))&(s>scipy.stats.norm.ppf(alpha/2)):
        print('Значение в контролькой группе равна тестовой группе.')
    elif s<scipy.stats.norm.ppf(alpha):
        print('Значение в контролькой группе меньше, чем в тестовой.')
    elif s>scipy.stats.norm.ppf(1-alpha):
        print('Значение в контролькой группе больше,чем в тестовой.')

def mann_whitney_func(r1, r2, alpha=0.05):
    s, p = mannwhitneyu(r1, r2)
    if p < alpha:
        print(f"Итоги теста:\nalpha = {alpha}\
\nНеобходимо отвергнуть нулевую гепотизу.Распределения не равны")

# Функция для вычисления ttest, pvalue_ttest, флаг (лейбл), по которому можно будет разделить торговые точки по результатам эксперимента.
def test_calc_2 (r1, r2, alpha = 0.05):
    s, p = ttest_ind(r1, r2)      
    if (s <scipy.stats.norm.ppf(1-alpha/2))&(s>scipy.stats.norm.ppf(alpha/2)):
        label_distinction = "нейтральный исход"
        return s, p , label_distinction
    elif s<scipy.stats.norm.ppf(alpha):
        label_distinction = "положительный исход"
        return s, p , label_distinction
    elif s>scipy.stats.norm.ppf(1-alpha):
        label_distinction = "отрицательный исход"
        return s, p , label_distinction

# Функция для расчёта sigma_test, sigma_control, ttest, pvalue_ttest, флаг (лейбл), по которому можно будет разделить торговые точки по результатам эксперимента.    
def result_ttest_ab_test(id_trading_point):

                                              (data_clients_region_agg['id_trading_point']==id_trading_point)]['amt_payment'],
                                         0.05)

    return round(s,4), round(p,4), label_distinction, round(sigma_test,4), round(sigma_control,4)
